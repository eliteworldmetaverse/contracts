pragma solidity ^0.8.2;
 
import "@openzeppelin/contracts/utils/math/SafeMath.sol";
import "./EliteWorldNFT.sol";
 
contract EliteWorld is EliteWorldNFT {
    using SafeMath for uint256;
 
    struct Item {
        string id;
        string tokenURI;
        string idType;
        uint256 price;
    }
 
    Item[] public items;
    mapping(string => uint256) public itemRegistry;
    mapping(string => bool) public freeItems;
    mapping(string => uint256) public itemTypeRegistry;
    mapping(address => bool) public allowBuyFree;
    mapping(string => address) private _itemOf;
    mapping(address => mapping(string => bool)) private _itemBlock;
 
    // ============ Modifiers ============
 
    modifier onlyAgent() {
        Erequire(msg.sender == agent, "Caller is not agent");
        _;
    }
 
    modifier itemExist(string memory _id) {
        Erequire(itemRegistry[_id] > 0, "Item Not Exist");
        _;
    }
 
    modifier itemNotExist(string memory _id) {
        Erequire(itemRegistry[_id] == 0, "Item Already Exist");
        _;
    }
 
    // ============ Helper ============
 
    function getItemId(string memory _id)
        public
        view
        itemExist(_id)
        returns (uint256)
    {
        return itemRegistry[_id] - 1;
    }
 
    function compareStringsbyBytes(string memory s1, string memory s2)
        internal
        pure
        returns (bool)
    {
        return
            keccak256(abi.encodePacked(s1)) == keccak256(abi.encodePacked(s2));
    }
 
    function getItemsLength() external view returns (uint256) {
        return items.length;
    }
 
    function getItemsByType(string memory _idType)
        external
        view
        returns (uint256[] memory)
    {
        uint256[] memory result = new uint256[](itemTypeRegistry[_idType]);
        uint256 counter = 0;
        for (uint256 i = 0; i < items.length; i++) {
            if (compareStringsbyBytes(items[i].idType, _idType)) {
                result[counter] = i;
                counter++;
            }
        }
        return result;
    }
 
    // ============ Ownable ============
 
    function setItem(
        string memory _id,
        string memory _tokenURI,
        string memory _idType,
        uint256 _price,
        bool _freeItem
    ) public itemNotExist(_id) onlyAgent {
        items.push(Item(_id, _tokenURI, _idType, _price));
        itemRegistry[_id] = items.length;
        itemTypeRegistry[_idType] = itemTypeRegistry[_idType].add(1);
        freeItems[_id] = _freeItem;
    }
 
    function editPriceItem(string memory _id, uint256 _price) public onlyAgent {
        uint256 id = getItemId(_id);
        Item storage item = items[id];
        item.price = _price;
    }
 
    function setFreeItem(string memory _id, bool _freeItem)
        public
        itemExist(_id)
        onlyAgent
    {
        freeItems[_id] = _freeItem;
    }
 
    // ============ Transaction ============
 
    // Allow buy 4 free item characters, shirt, pants, shoes at the same time
    // Buy only once
    // Free items cannot be resold or auctioned
    function buyFreeItems(
        string memory _id0,
        string memory _id1,
        string memory _id2,
        string memory _id3
    ) external {
        require(allowBuyFree[msg.sender] != true, "Free item purchased");
        Erequire(freeItems[_id0] == true, "Invalid item characters");
        Erequire(freeItems[_id1] == true, "Invalid item shirt");
        Erequire(freeItems[_id2] == true, "Invalid item pants");
        Erequire(freeItems[_id3] == true, "Invalid item shoes");
 
        uint256 id0 = getItemId(_id0);
        uint256 id1 = getItemId(_id1);
        uint256 id2 = getItemId(_id2);
        uint256 id3 = getItemId(_id3);
        Erequire(
            compareStringsbyBytes(items[id0].idType, "characters"),
            "Required characters item"
        );
        Erequire(
            compareStringsbyBytes(items[id1].idType, "shirt"),
            "Required shirt item"
        );
        Erequire(
            compareStringsbyBytes(items[id1].idType, "shirt"),
            "Required pants item"
        );
        Erequire(
            compareStringsbyBytes(items[id1].idType, "shirt"),
            "Required shoes item"
        );
 
        _mintItem(msg.sender, items[id0].tokenURI);
        _itemOf[items[id0].id] = msg.sender;
        _itemBlock[msg.sender][items[id0].id] = true;
 
        _mintItem(msg.sender, items[id1].tokenURI);
        _itemOf[items[id1].id] = msg.sender;
        _itemBlock[msg.sender][items[id1].id] = true;
 
        _mintItem(msg.sender, items[id2].tokenURI);
        _itemOf[items[id2].id] = msg.sender;
        _itemBlock[msg.sender][items[id2].id] = true;
 
        _mintItem(msg.sender, items[id3].tokenURI);
        _itemOf[items[id3].id] = msg.sender;
        _itemBlock[msg.sender][items[id3].id] = true;
 
        allowBuyFree[msg.sender] = true;
    }
 
    // Only buy items that you don't own
    function buyItem(string memory _id) external {
        uint256 id = getItemId(_id);
        Item memory item = items[id];
        Erequire(_itemOf[item.id] != msg.sender, "Item purchased");
 
        IBunVault(bunVault).deposit(address(msg.sender), item.price);
 
        _mintItem(msg.sender, item.tokenURI);
    }
