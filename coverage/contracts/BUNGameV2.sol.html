pragma solidity ^0.8.0;
 
import "@openzeppelin/contracts-upgradeable/access/OwnableUpgradeable.sol";
import "@openzeppelin/contracts/utils/math/SafeMath.sol";
import "@openzeppelin/contracts-upgradeable/token/ERC20/utils/SafeERC20Upgradeable.sol";
import "@openzeppelin/contracts-upgradeable/proxy/utils/Initializable.sol";
 
interface IBunVault {
  function initialize(address bunToken) external;
 
  function deposit(address from, uint256 amount) external;
 
  function withdraw(address to, uint256 amount) external;
}
 
contract BunRewardVault is Initializable, OwnableUpgradeable {
  using SafeERC20Upgradeable for IERC20Upgradeable;
 
  address public bunToken;
 
  function initialize(address _bunToken) public initializer {
    bunToken = _bunToken;
    __Ownable_init();
  }
 
  function withdraw(address to, uint256 amount) external onlyOwner {
    IERC20Upgradeable(bunToken).safeTransfer(to, amount);
  }
 
  function deposit(address from, uint256 amount) external onlyOwner {
    IERC20Upgradeable(bunToken).safeTransferFrom(from, address(this), amount);
  }
}
 
contract BUNGameV2 is Initializable, OwnableUpgradeable {
  using SafeMath for uint256;
  using SafeERC20Upgradeable for IERC20Upgradeable;
 
  address public bunVault;
  address private agent;
  uint256 private fee;
  uint256 public gameFee;
 
  mapping(address => uint256) private _balances;
 
  struct TypeRoomInfo {
    string name;
    uint256 bet;
  }
 
  TypeRoomInfo[] public typeRoomInfos;
 
  struct Room {
    uint256 rid;
    address[] players;
    uint8 status;
  }
 
  mapping(string => Room) private rooms;
 
  struct InfoUser {
    uint256 deposit;
    uint256 win;
    uint256 lose;
  }
 
  mapping(address => InfoUser) public infoUsers;
 
  event Deposit(address indexed account, uint256 amount);
  event Withdraw(address indexed user, uint256 amount);
  event WithdrawFee(address indexed user, uint256 amount);
  event AgentSet(address indexed oldAgent, address indexed newAgent);
  event TypeRoomInfoCreate(uint256 roomId);
  event GameStart(string roomId);
  event GameEnd(string roomId, address indexed winner);
  event ForceGameEnd(string roomId);
 
  function initialize(
    address _bunVault,
    address _agent,
    uint256 _gameFee
  ) public initializer {
    bunVault = address(new BunRewardVault());
    IBunVault(bunVault).initialize(_bunVault);
    agent = _agent;
    gameFee = _gameFee;
    __Ownable_init();
  }
 
  // ============ Modifiers ============
  modifier onlyAgent() {
    require(msg.sender == agent, "Caller is not agent");
    _;
  }
 
  // ============ Ownable ============
 
  /*
   * @dev create type room
   */
  function addTypeRoom(string memory _name, uint256 _bet) public onlyAgent {
    typeRoomInfos.push(TypeRoomInfo(_name, _bet));
    emit TypeRoomInfoCreate(typeRoomInfos.length - 1);
  }
 
  function setTypeRoom(
    uint256 _rid,
    string memory _name,
    uint256 _bet
  ) public onlyAgent {
    typeRoomInfos[_rid].name = _name;
    typeRoomInfos[_rid].bet = _bet;
  }
 
  function totalTypeRoom() public view returns (uint256) {
    return typeRoomInfos.length;
  }
 
  /**
   * @dev set agent
   * @param _agent address of new agent
   */
  function setAgent(address _agent) public onlyOwner {
    address oldAgent = agent;
    agent = _agent;
    emit AgentSet(oldAgent, _agent);
  }
 
  /**
   * @dev Set gameFee
   * @param _gameFee percent of reward will be deducted
   */
  function setGameFee(uint256 _gameFee) public onlyAgent {
    require(_gameFee > 0 && _gameFee < 100, "GameFee must be from 1 to 99");
    gameFee = _gameFee;
  }
 
  /*
   * @dev get balance of account
   */
  function balanceOf(address _account) public view returns (uint256) {
    return _balances[_account];
  }
 
  /*
   * @dev deposit
   * @param _amount amount to deposit
   */
  function deposit(uint256 _amount) public {
    IBunVault(bunVault).deposit(address(msg.sender), _amount);
 
    _balances[msg.sender] = _balances[msg.sender].add(_amount);
 
    InfoUser storage user = infoUsers[msg.sender];
    user.deposit = user.deposit.add(_amount);
 
    emit Deposit(msg.sender, _amount);
  }
 
  /*
   * @dev withdraw balance
   * @param _amount amount to withdraw
   */
  function withdraw(uint256 _amount) public {
    require(
      _balances[msg.sender] >= _amount,
      "withdraw amount exceeds balance"
    );
    IBunVault(bunVault).withdraw(address(msg.sender), _amount);
    _balances[msg.sender] = _balances[msg.sender].sub(_amount);
 
    emit Withdraw(msg.sender, _amount);
  }
 
  /*
   * @dev get amount fee
   */
  function getFee() public view onlyOwner returns (uint256) {
    return fee;
  }
 
  /*
   * @dev withdraw fee from game
   * @param to: address to withdraw
   */
  function withdrawFee(address _to) public onlyOwner {
    require(fee > 0, "Empty Fee");
    IBunVault(bunVault).withdraw(_to, fee);
 
    emit WithdrawFee(msg.sender, fee);
  }
 
  /*
   * @dev start game
   * @param _gameId Id of newGame, _gametype type of newGame, _players addresses of players
   */
  function startGame(
    string memory _roomId,
    uint256 _rid,
    address[] memory _players
  ) public onlyAgent {
    require(_rid < typeRoomInfos.length, "Type Game is not exists");
    require(_players.length > 1, "Minimum 2 players");
    require(
      rooms[_roomId].status != 1 && rooms[_roomId].status != 2,
      "Game is playing or end"
    );
 
    TypeRoomInfo memory typeRoomInfo = typeRoomInfos[_rid];
    /**
     * checking if all player have enough money
     */
    for (uint8 i = 0; i < _players.length; i++) {
      require(
        _balances[_players[i]] >= typeRoomInfo.bet,
        "Player exceeds balance"
      );
    }
 
    for (uint8 i = 0; i < _players.length; i++) {
      _balances[_players[i]] = _balances[_players[i]].sub(typeRoomInfo.bet);
    }
 
    Room storage room = rooms[_roomId];
    room.status = 1;
    room.players = _players;
    room.rid = _rid;
 
    emit GameStart(_roomId);
  }
 
  /**
   * @dev end game to reward
   */
 
  function endGame(string memory _roomId, address _winner) public onlyAgent {
    Room storage room = rooms[_roomId];
    require(room.status == 1, "Game is not exists");
 
    bool isWinnerExist = false;
    for (uint256 i = 0; i < room.players.length; i++) {
      if (room.players[i] == _winner) {
        isWinnerExist = true;
      }
    }
 
    require(isWinnerExist, "Invalid winner");
    TypeRoomInfo memory typeRoomInfo = typeRoomInfos[room.rid];
 
    uint256 feeInGame = typeRoomInfo.bet.mul(room.players.length).div(100).mul(
      gameFee
    );
    uint256 reward = typeRoomInfo.bet.mul(room.players.length).sub(feeInGame);
 
    fee = fee.add(feeInGame);
    _balances[_winner] = _balances[_winner].add(reward);
 
    for (uint256 i = 0; i < room.players.length; i++) {
      if (room.players[i] == _winner) {
        infoUsers[room.players[i]].win = infoUsers[room.players[i]].win.add(
          reward.sub(typeRoomInfo.bet)
        );
      } else {
        infoUsers[room.players[i]].lose = infoUsers[room.players[i]].lose.add(
          typeRoomInfo.bet
        );
      }
    }
 
    room.status = 2;
    emit GameEnd(_roomId, _winner);
  }
 
  /**
   * @dev end game when user disconnect
   */
 
  function forceEndGame(string memory _roomId) public onlyAgent {
    Room memory room = rooms[_roomId];
    require(room.status == 1, "Game is not exists");
    TypeRoomInfo memory typeRoomInfo = typeRoomInfos[room.rid];
 
    fee = fee.add(typeRoomInfo.bet.mul(room.players.length).div(2));
 
    for (uint256 i = 0; i < room.players.length; i++) {
      infoUsers[room.players[i]].lose = infoUsers[room.players[i]].lose.add(
        typeRoomInfo.bet.div(2)
      );
      _balances[room.players[i]] = _balances[room.players[i]].add(
        typeRoomInfo.bet.div(2)
      );
    }
 
    delete rooms[_roomId];
    emit ForceGameEnd(_roomId);
  }
}
